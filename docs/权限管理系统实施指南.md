```csharp
using TcmAiDiagnosis.EFContext;

            AddPermissionGroup(permissions, "用户管理", "UserManagement", typeof(PermissionConstants.UserManagement), now);

            AddPermissionGroup(permissions, "租户监控", "TenantMonitoring", typeof(PermissionConstants.TenantMonitoring), now);

            AddPermissionGroup(permissions, "个人信息", "PersonalInformation", typeof(PermissionConstants.PersonalInformation), now);

            AddPermissionGroup(permissions, "治疗方案查看", "TreatmentPlanView", typeof(PermissionConstants.TreatmentPlanView), now);

            AddPermissionGroup(permissions, "系统维护", "SystemMaintenance", typeof(PermissionConstants.SystemMaintenance), now);

            await context.Permissions.AddRangeAsync(permissions);
            await context.SaveChangesAsync();
        }

        private static void AddPermissionGroup(List<Permission> permissions, string category, string module, Type constantsType, DateTime now)
        {
            var fields = constantsType.GetFields(System.Reflection.BindingFlags.Public | System.Reflection.BindingFlags.Static);
            int sortOrder = 0;

            foreach (var field in fields)
            {
                if (field.FieldType == typeof(string))
                {
                    var code = field.GetValue(null)?.ToString();
                    if (!string.IsNullOrEmpty(code))
                    {
                        permissions.Add(new Permission
                        {
                            PermissionCode = code,
                            PermissionName = GetPermissionName(field.Name),
                            Description = $"{category} - {GetPermissionName(field.Name)}",
                            Category = category,
                            Module = module,
                            IsSystem = true,
                            IsActive = true,
                            SortOrder = sortOrder++,
                            CreatedAt = now,
                            UpdatedAt = now
                        });
                    }
                }
            }
        }

        private static string GetPermissionName(string fieldName)
        {
            // 将驼峰命名转换为中文友好的名称
            var nameMap = new Dictionary<string, string>
            {
                { "Create", "创建" },
                { "Edit", "编辑" },
                { "View", "查看" },
                { "Delete", "删除" },
                { "Enable", "启用" },
                { "Disable", "禁用" },
                { "Search", "搜索" },
                { "Export", "导出" },
                { "Print", "打印" },
                { "Sign", "签名" },
                { "Confirm", "确认" },
                { "Review", "审核" },
                // 添加更多映射...
            };

            return nameMap.ContainsKey(fieldName) ? nameMap[fieldName] : fieldName;
        }
    }
}
```

```csharp
using TcmAiDiagnosis.EFContext;
using TcmAiDiagnosis.Entities;
using TcmAiDiagnosis.Domain.Constants;
using Microsoft.EntityFrameworkCore;
using Microsoft.AspNetCore.Identity;

namespace TcmAiDiagnosis.Web.Data
{
    /// <summary>
    /// 角色权限数据种子
    /// </summary>
    public static class RolePermissionSeeder
    {
        public static async Task SeedRolePermissionsAsync(
            TcmAiDiagnosisContext context,
            RoleManager<Role> roleManager,
            UserManager<User> userManager)
        {
            // 检查是否已有角色权限数据
            if (await context.RolePermissions.AnyAsync())
            {
                return; // 已有数据，跳过初始化
            }

            var now = DateTime.UtcNow;
            var adminUser = await userManager.FindByNameAsync("admin");
            var adminUserId = adminUser?.Id ?? 0;

            // 获取所有权限
            var allPermissions = await context.Permissions.ToListAsync();
            var permissionDict = allPermissions.ToDictionary(p => p.PermissionCode, p => p.PermissionId);

            // 1. 平台超级管理员角色权限
            var managerRole = await roleManager.FindByNameAsync("Manager");
            if (managerRole != null)
            {
                await AssignPermissionsToRole(context, managerRole.Id, GetManagerPermissions(), permissionDict, adminUserId, now);
            }

            // 2. 医生角色权限
            var doctorRole = await roleManager.FindByNameAsync("Doctor");
            if (doctorRole != null)
            {
                await AssignPermissionsToRole(context, doctorRole.Id, GetDoctorPermissions(), permissionDict, adminUserId, now);
            }

            // 3. 患者角色权限
            var patientRole = await roleManager.FindByNameAsync("Patient");
            if (patientRole != null)
            {
                await AssignPermissionsToRole(context, patientRole.Id, GetPatientPermissions(), permissionDict, adminUserId, now);
            }

            await context.SaveChangesAsync();
        }

        private static async Task AssignPermissionsToRole(
            TcmAiDiagnosisContext context,
            int roleId,
            List<string> permissionCodes,
            Dictionary<string, int> permissionDict,
            int grantedBy,
            DateTime grantedAt)
        {
            var rolePermissions = new List<RolePermission>();

            foreach (var code in permissionCodes)
            {
                if (permissionDict.TryGetValue(code, out int permissionId))
                {
                    rolePermissions.Add(new RolePermission
                    {
                        RoleId = roleId,
                        PermissionId = permissionId,
                        IsGranted = true,
                        GrantedAt = grantedAt,
                        GrantedBy = grantedBy > 0 ? grantedBy : null,
                        Remarks = "系统初始化分配"
                    });
                }
            }

            await context.RolePermissions.AddRangeAsync(rolePermissions);
        }
```

## 第三阶段：业务逻辑层实现

需要在 `TcmAiDiagnosis.Domain` 项目中创建以下 Domain 类：

### 3.1 权限管理 Domain

- `PermissionDomain.cs` - 权限CRUD操作

### 3.2 实现要点

1. **权限验证逻辑**
   - 检查用户是否拥有指定权限
   - 支持角色继承
   - 支持临时权限
   - 支持权限覆盖（拒绝优先）

2. **审计日志**
   - 所有权限变更必须记录
   - 记录操作人、时间、IP、原因等

3. **临时权限管理**
   - 自动过期检查
   - 访问次数限制
   - 患者授权和管理员审批流程

```csharp

GET    /api/permissions              // 获取权限列表
```

## 常见问题

### Q1: 迁移失败，提示文件被锁定

**A:** 停止所有运行中的应用程序实例，包括：
- Visual Studio 调试进程
- IIS Express
- 命令行运行的 dotnet run
- 后台运行的进程

### Q2: 权限数据初始化失败

**A:** 检查：
- 数据库连接是否正常
- PermissionConstants 类是否正确定义
- 是否有重复的权限代码

### Q3: 权限验证不生效

**A:** 确认：
- 中间件是否正确注册
- 中间件顺序是否正确（在 UseAuthorization 之后）
- 用户是否正确分配了角色和权限
