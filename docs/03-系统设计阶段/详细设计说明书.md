**核心服务: `TreatmentDomain`**

```csharp
public class TreatmentDomain : ITreatmentDomain
{
    private readonly IVisitRepository _visitRepository;
    private readonly ITreatmentRepository _treatmentRepository;
    private readonly ICompatibilityRuleEngine _ruleEngine;
    private readonly IMapper _mapper;
    private readonly ILogger<TreatmentDomain> _logger;

    public TreatmentDomain(
        IVisitRepository visitRepository,
        ITreatmentRepository treatmentRepository,
        ICompatibilityRuleEngine ruleEngine,
        IMapper mapper,
        ILogger<TreatmentDomain> logger)
    {
        _visitRepository = visitRepository;
        _treatmentRepository = treatmentRepository;
        _ruleEngine = ruleEngine;
        _mapper = mapper;
        _logger = logger;
    }

    public async Task<TreatmentDto> CreateTreatmentAsync(CreateTreatmentRequest request)
    {
        // 1. 验证就诊记录是否存在
        var visit = await _visitRepository.GetByIdAsync(request.VisitId) 
                    ?? throw new NotFoundException($"就诊记录 (ID: {request.VisitId}) 不存在。");

        // 2. 创建治疗方案聚合根
        var treatment = Treatment.Create(
            request.VisitId,
            visit.DoctorId,
            visit.PatientId,
            request.Principle,
            request.Analysis
        );

        // 3. 处理中药处方
        if (request.Prescription != null)
        {
            // 3.1 执行配伍禁忌检查
            var compatibilityResult = await _ruleEngine.CheckAsync(request.Prescription.Items);
            if (compatibilityResult.HasConflict && !request.ForceSave)
            {
                throw new BusinessException(
                    $"存在配伍禁忌: {compatibilityResult.Message}。如需强制保存，请确认。",
                    "COMPATIBILITY_CONFLICT"
                );
            }
            
            // 3.2 添加处方到治疗方案
            treatment.AddPrescription(
                request.Prescription.PrescriptionName,
                request.Prescription.Dosage,
                request.Prescription.Usage,
                request.Prescription.BrewingMethod,
                request.Prescription.SpecialInstructions,
                request.Prescription.Items.Select(item => 
                    new PrescriptionItem(item.HerbId, item.HerbName, item.Dosage, item.Unit, item.Notes))
            );
        }

        // 4. 处理非药物疗法
        request.NonDrugTherapies?.ForEach(therapy =>
            treatment.AddNonDrugTherapy(therapy.TherapyType, therapy.Description, therapy.Usage)
        );

        // 5. 处理医嘱
        request.Advices?.ForEach(advice =>
            treatment.AddAdvice(advice.AdviceType, advice.Content)
        );
        
        // 6. 处理风险警告
        request.Warnings?.ForEach(warning =>
            treatment.AddWarning(warning.WarningType, warning.Title, warning.Content)
        );

        // 7. 持久化
        await _treatmentRepository.AddAsync(treatment);
        _logger.LogInformation("治疗方案创建成功 - TreatmentId: {TreatmentId}, VisitId: {VisitId}", treatment.Id, visit.Id);

        return _mapper.Map<TreatmentDto>(treatment);
    }
}
```

### 2.1.4 算法设计：配伍禁忌检查

**伪代码: `CompatibilityRuleEngine`**

```csharp
public class CompatibilityRuleEngine : ICompatibilityRuleEngine
{
    // “十八反”规则库，可以从数据库或配置文件加载
    private readonly Dictionary<string, string[]> EighteenAntagonisms = new Dictionary<string, string[]>
    {
        { "甘草", new[] { "甘遂", "大戟", "海藻", "芫花" } },
        { "乌头", new[] { "贝母", "瓜蒌", "半夏", "白蔹", "白及" } },
        // ... 其他规则
    };

    public CompatibilityResult Check(List<PrescriptionItemDto> items)
    {
        var itemNames = items.Select(i => i.HerbName).ToHashSet();

        foreach (var item in items)
        {
            // 检查十八反
            if (EighteenAntagonisms.ContainsKey(item.HerbName))
            {
                foreach (var antagonist in EighteenAntagonisms[item.HerbName])
                {
                    if (itemNames.Contains(antagonist))
                    {
                        return new CompatibilityResult 
                        {
                            HasConflict = true, 
                            Message = $"【十八反】'{item.HerbName}' 与 '{antagonist}' 相反。"
                        };
                    }
                }
            }

            // 检查“十九畏”等其他规则...
        }

        return CompatibilityResult.Success;
    }
}

### 2.2 用户管理模块

#### 2.1.1 患者管理子模块

**概述**
患者管理子模块负责处理患者的基础信息管理，包括患者注册、信息更新、查询等功能。该模块支持跨租户患者信息共享，确保患者在不同医疗机构间的信息一致性，同时保证病历数据的租户隔离。

**核心设计原则**
1. **信息共享**：患者基础信息（姓名、手机号、身份证等）在租户间共享
2. **数据隔离**：病历数据严格按租户隔离，互不可见
3. **关联管理**：通过患者-租户关联表管理患者在不同租户的关系
4. **权限控制**：只有授权医生才能搜索和添加跨租户患者

##### 类设计

```csharp
public class PatientService : IPatientService
{
    private readonly IPatientRepository _patientRepository;
    private readonly IUserRepository _userRepository;
    private readonly IPatientTenantRepository _patientTenantRepository;
    private readonly ITenantContext _tenantContext;
    private readonly ICrossTenantPatientService _crossTenantPatientService;
    private readonly IMapper _mapper;
    private readonly ILogger<PatientService> _logger;
    
    public async Task<PatientDto> CreatePatientAsync(CreatePatientRequest request)
    {
        // 1. 验证患者信息
        await ValidatePatientInfoAsync(request);
        
        // 2. 检查是否已存在相同手机号的患者
        var existingPatient = await _patientRepository.GetByPhoneAsync(request.Phone);
        if (existingPatient != null)
        {
            // 如果患者已存在但未关联到当前租户，提示可以添加关联
            var isLinkedToCurrentTenant = await _patientTenantRepository
                .IsPatientLinkedToTenantAsync(existingPatient.Id, _tenantContext.TenantId);
            
            if (!isLinkedToCurrentTenant)
            {
                throw new BusinessException($"该手机号患者已存在，患者ID: {existingPatient.Id}，可通过添加现有患者功能关联到当前租户");
            }
            else
            {
                throw new BusinessException("该手机号已在当前租户注册");
            }
        }
        
        // 3. 创建用户账户
        var user = new User
        {
            Phone = request.Phone,
            UserType = UserType.Patient,
            CreatedAt = DateTime.UtcNow
        };
        await _userRepository.CreateAsync(user);
        
        // 4. 创建患者信息
        var patient = _mapper.Map<Patient>(request);
        patient.UserId = user.Id;
        patient.CreatedAt = DateTime.UtcNow;
        
        await _patientRepository.CreateAsync(patient);
        
        // 5. 创建患者-租户关联关系
        var patientTenant = new PatientTenant
        {
            PatientId = patient.Id,
            TenantId = _tenantContext.TenantId,
            AddedBy = request.CreatedBy,
            AddedAt = DateTime.UtcNow,
            IsActive = true,
            RelationType = PatientTenantRelationType.Original // 原始注册
        };
        await _patientTenantRepository.CreateAsync(patientTenant);
        
        _logger.LogInformation("患者创建成功 - 患者ID: {PatientId}, 租户ID: {TenantId}", 
            patient.Id, _tenantContext.TenantId);
            
        return _mapper.Map<PatientDto>(patient);
    }
    
    public async Task<PagedResult<PatientDto>> SearchPatientsAsync(SearchPatientsRequest request)
    {
        // 获取租户关联的所有患者
        var patients = await _patientRepository.GetPatientsByTenantAsync(_tenantContext.TenantId, request);
        return new PagedResult<PatientDto>
        {
            Items = _mapper.Map<List<PatientDto>>(patients.Items),
            TotalCount = patients.TotalCount,
            PageIndex = patients.PageIndex,
            PageSize = patients.PageSize
        };
    }
    
    public async Task<PatientDto> GetPatientByIdAsync(int patientId)
    {
        // 验证患者是否关联到当前租户
        var isLinked = await _patientTenantRepository.IsPatientLinkedToTenantAsync(patientId, _tenantContext.TenantId);
        if (!isLinked)
        {
            throw new UnauthorizedException("无权限访问该患者信息");
        }
        
        var patient = await _patientRepository.GetByIdAsync(patientId);
        if (patient == null)
        {
            throw new NotFoundException("患者不存在");
        }
        
        return _mapper.Map<PatientDto>(patient);
    }
    
    public async Task<bool> UpdatePatientAsync(int patientId, UpdatePatientRequest request)
    {
        // 验证患者是否关联到当前租户
        var isLinked = await _patientTenantRepository.IsPatientLinkedToTenantAsync(patientId, _tenantContext.TenantId);
        if (!isLinked)
        {
            throw new UnauthorizedException("无权限访问该患者信息");
        }
        
        var patient = await _patientRepository.GetByIdAsync(patientId);
        if (patient == null)
        {
            throw new NotFoundException("患者不存在");
        }
        
        // 更新患者基础信息（全局共享）
        _mapper.Map(request, patient);
        patient.UpdatedAt = DateTime.UtcNow;
        
        await _patientRepository.UpdateAsync(patient);
        
        _logger.LogInformation("患者信息更新成功 - 患者ID: {PatientId}, 租户ID: {TenantId}", 
            patientId, _tenantContext.TenantId);
        
        return true;
    }
    
    public async Task<PatientDto> SearchPatientByPhoneAsync(string phoneNumber)
    {
        // 跨租户搜索患者逻辑
        return await _crossTenantPatientService.SearchPatientByPhoneAsync(phoneNumber);
    }
    
    public async Task<bool> AddExistingPatientAsync(AddExistingPatientRequest request)
    {
        // 添加已存在患者到当前租户逻辑
        return await _crossTenantPatientService.AddExistingPatientToTenantAsync(request);
    }
}

public class CrossTenantPatientService : ICrossTenantPatientService
{
    private readonly IPatientRepository _patientRepository;
    private readonly IPatientTenantRepository _patientTenantRepository;
    private readonly IDoctorPatientRepository _doctorPatientRepository;
    private readonly ITenantRepository _tenantRepository;
    private readonly ITenantContext _tenantContext;
    private readonly ICurrentUser _currentUser;
    private readonly IMapper _mapper;
    private readonly ILogger<CrossTenantPatientService> _logger;
    
    public async Task<List<PatientSearchResultDto>> SearchPatientByPhoneAsync(string phoneNumber)
    {
        // 1. 验证手机号格式
        if (!IsValidPhoneNumber(phoneNumber))
        {
            throw new ValidationException("手机号格式不正确");
        }
        
        // 2. 根据手机号搜索所有患者
        var patients = await _patientRepository.SearchByPhoneAsync(phoneNumber);
        
        if (!patients.Any())
        {
            return new List<PatientSearchResultDto>();
        }
        
        // 3. 获取当前租户已关联的患者ID列表
        var currentTenantPatientIds = await _patientTenantRepository
            .GetPatientIdsByTenantAsync(_tenantContext.TenantId);
        
        // 4. 构建搜索结果
        var results = new List<PatientSearchResultDto>();
        
        foreach (var patient in patients)
        {
            var isAlreadyLinked = currentTenantPatientIds.Contains(patient.Id);
            
            // 获取患者的原始租户信息
            var originalTenant = await GetPatientOriginalTenantAsync(patient.Id);
            
            var result = new PatientSearchResultDto
            {
                PatientId = patient.Id,
                Name = patient.Name,
                Phone = patient.Phone,
                Gender = patient.Gender,
                BirthDate = patient.BirthDate,
                IdCard = MaskIdCard(patient.IdCard), // 脱敏处理
                OriginalTenantName = originalTenant?.Name,
                IsAlreadyLinked = isAlreadyLinked,
                CanAddToCurrentTenant = !isAlreadyLinked,
                LinkedTenantCount = await GetPatientLinkedTenantCountAsync(patient.Id)
            };
            
            results.Add(result);
        }
        
        _logger.LogInformation("跨租户患者搜索完成 - 手机号: {Phone}, 找到患者数: {Count}, 可添加数: {AvailableCount}", 
            phoneNumber, results.Count, results.Count(r => r.CanAddToCurrentTenant));
            
        return results;
    }
    
    public async Task<bool> AddExistingPatientToTenantAsync(AddExistingPatientRequest request)
    {
        // 1. 验证患者是否存在
        var patient = await _patientRepository.GetByIdAsync(request.PatientId);
        if (patient == null)
            throw new BusinessException("患者不存在");
            
        // 2. 检查是否已关联到目标租户
        var existingRelation = await _patientTenantRepository
            .GetByPatientAndTenantAsync(request.PatientId, _tenantContext.TenantId);
        if (existingRelation != null)
        {
            if (existingRelation.IsActive)
            {
                throw new BusinessException("患者已关联到该租户");
            }
            else
            {
                // 重新激活已停用的关联关系
                existingRelation.IsActive = true;
                existingRelation.ReactivatedBy = _currentUser.UserId;
                existingRelation.ReactivatedAt = DateTime.UtcNow;
                await _patientTenantRepository.UpdateAsync(existingRelation);
                
                _logger.LogInformation("患者关联关系重新激活 - 患者ID: {PatientId}, 租户ID: {TenantId}", 
                    request.PatientId, _tenantContext.TenantId);
            }
        }
        else
        {
            // 3. 创建新的患者-租户关联关系
            var patientTenant = new PatientTenant
            {
                PatientId = request.PatientId,
                TenantId = _tenantContext.TenantId,
                AddedBy = _currentUser.UserId,
                AddedAt = DateTime.UtcNow,
                IsActive = true,
                RelationType = PatientTenantRelationType.CrossTenant // 跨租户添加
            };
            
            await _patientTenantRepository.AddAsync(patientTenant);
            
            _logger.LogInformation("患者成功关联到租户 - 患者ID: {PatientId}, 租户ID: {TenantId}, 操作人: {OperatorId}", 
                request.PatientId, _tenantContext.TenantId, _currentUser.UserId);
        }
        
        // 4. 建立医生-患者关联
        await _doctorPatientRepository.AddAsync(new DoctorPatientJunction
        {
            DoctorId = _currentUser.UserId,
            PatientId = request.PatientId,
            TenantId = _tenantContext.TenantId,
            CreatedAt = DateTime.UtcNow
        });
        
        return true;
    }
    
    public async Task RemovePatientFromTenantAsync(int patientId, int tenantId, int operatorId)
    {
        var relation = await _patientTenantRepository.GetByPatientAndTenantAsync(patientId, tenantId);
        if (relation == null || !relation.IsActive)
        {
            throw new NotFoundException("患者租户关联关系不存在");
        }
        
        // 检查是否为原始租户
        if (relation.RelationType == PatientTenantRelationType.Original)
        {
            throw new BusinessException("不能移除患者的原始租户关联");
        }
        
        // 软删除关联关系
        relation.IsActive = false;
        relation.RemovedBy = operatorId;
        relation.RemovedAt = DateTime.UtcNow;
        
        await _patientTenantRepository.UpdateAsync(relation);
        
        _logger.LogInformation("患者从租户移除 - 患者ID: {PatientId}, 租户ID: {TenantId}, 操作人: {OperatorId}",

### 2.2 AI辅助诊断模块

#### 2.2.1 模块概述

AI辅助诊断模块是本系统的核心特色功能，旨在利用大语言模型（LLM）辅助医生进行证候分析和诊断。本模块通过与第三方AI平台Dify的集成，将医生录入的患者四诊信息发送给预设的AI工作流，获取结构化的证候分析结果，包括可能的证候、辨证依据、治法方药建议等。

#### 2.2.2 核心流程

1.  **信息采集**：医生在诊疗界面录入患者的主诉、现病史、舌象、脉象等四诊信息。
2.  **数据整合**：系统将采集到的信息整合成结构化的文本描述。
3.  **API调用**：
    *   **证候概览**：调用Dify的“证候概览”工作流，发送简要的患者信息和问诊信息，快速获取一个或多个可能性较高的证候列表。
    *   **证候详情**：当医生选择某个具体证候时，调用Dify的“证候详情”工作流，传入该证候名称及完整的四诊信息，获取该证候的详细分析，包括辨证要点、相似证候鉴别、推荐治法、经典方剂等。
4.  **结果展示**：系统解析API返回的JSON数据，并在前端以清晰、直观的方式展示给医生，作为临床决策的参考。
5.  **诊断采纳**：医生可以参考AI分析结果，结合自己的专业判断，选择一个或多个证候作为本次的诊断结论，并将其存入就诊记录。

    public async Task<VisitSeriesDto> CreateVisitSeriesAsync(CreateVisitSeriesRequest request)
    {
        // 检查患者是否有未结束的就诊系列
        var activeSeriesList = await _visitSeriesRepository
            .GetActiveSeriesByPatientAsync(request.PatientId, _tenantContext.TenantId);
            
        if (activeSeriesList.Any())
        {
            // 根据业务规则自动结束旧的就诊系列
            foreach (var activeSeries in activeSeriesList)
            {
                await EndVisitSeriesAsync(activeSeries.Id, new EndVisitSeriesRequest
                {
                    EndReason = "系统自动结束：创建新就诊系列",
                    EndedByDoctorId = _currentUser.UserId
                });
            }
        }
        
        // 创建新的就诊系列
        var newSeries = new VisitSeries
        {
            PatientId = request.PatientId,
            DoctorId = _currentUser.UserId,
            TenantId = _tenantContext.TenantId,
            ChiefComplaint = request.ChiefComplaint,
            Status = VisitSeriesStatus.Active,
            StartDate = DateTime.UtcNow,
            CreatedAt = DateTime.UtcNow
        };
        
        await _visitSeriesRepository.AddAsync(newSeries);
        return MapToDto(newSeries);
    }
    
    public async Task<VisitSeriesDto> GetActiveSeriesByPatientAsync(int patientId)
    {
        var activeSeries = await _visitSeriesRepository
            .GetActiveSeriesByPatientAsync(patientId, _tenantContext.TenantId);
        return activeSeries.FirstOrDefault()?.MapToDto();
    }
    
    public async Task<bool> EndVisitSeriesAsync(int seriesId, EndVisitSeriesRequest request)
    {
        var series = await _visitSeriesRepository.GetByIdAsync(seriesId);
        if (series == null || series.TenantId != _tenantContext.TenantId)
            throw new BusinessException("就诊系列不存在");
            
        if (series.Status != VisitSeriesStatus.Active)
            throw new BusinessException("就诊系列已结束");
            
        series.Status = VisitSeriesStatus.Ended;
        series.EndDate = DateTime.UtcNow;
        series.EndReason = request.EndReason;
        series.EndedByDoctorId = request.EndedByDoctorId;
        
        await _visitSeriesRepository.UpdateAsync(series);
        return true;
    }
    
    public async Task<List<VisitSeriesDto>> GetPatientVisitSeriesAsync(int patientId)
    {
        var seriesList = await _visitSeriesRepository
            .GetByPatientIdAsync(patientId, _tenantContext.TenantId);
        return seriesList.Select(MapToDto).ToList();
    }

##### 2.3.1.2 就诊系列自动结束后台服务

```csharp
public class VisitSeriesAutoEndBackgroundService : BackgroundService
{
    private readonly IServiceProvider _serviceProvider;
    private readonly ILogger<VisitSeriesAutoEndBackgroundService> _logger;
    private readonly IConfiguration _configuration;
    
    // 可配置的参数
    private readonly int _checkIntervalMinutes;
    private readonly int _autoEndDaysThreshold;
    private readonly bool _isEnabled;
    
    public VisitSeriesAutoEndBackgroundService(
        IServiceProvider serviceProvider,
        ILogger<VisitSeriesAutoEndBackgroundService> logger,
        IConfiguration configuration)
    {
        _serviceProvider = serviceProvider;
        _logger = logger;
        _configuration = configuration;
        
        // 从配置文件读取参数
        _checkIntervalMinutes = _configuration.GetValue<int>("VisitSeriesAutoEnd:CheckIntervalMinutes", 60);
        _autoEndDaysThreshold = _configuration.GetValue<int>("VisitSeriesAutoEnd:AutoEndDaysThreshold", 30);
        _isEnabled = _configuration.GetValue<bool>("VisitSeriesAutoEnd:Enabled", true);
    }
    
    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        if (!_isEnabled)
        {
            _logger.LogInformation("就诊系列自动结束服务已禁用");
            return;
        }
        
        _logger.LogInformation("就诊系列自动结束服务已启动，检查间隔：{Interval}分钟，自动结束阈值：{Threshold}天", 
            _checkIntervalMinutes, _autoEndDaysThreshold);
            
        while (!stoppingToken.IsCancellationRequested)
        {
            try
            {
                await ProcessAutoEndVisitSeriesAsync();
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "执行就诊系列自动结束任务时发生错误");
            }
            
            await Task.Delay(TimeSpan.FromMinutes(_checkIntervalMinutes), stoppingToken);
        }
    }
    
    private async Task ProcessAutoEndVisitSeriesAsync()
    {
        using var scope = _serviceProvider.CreateScope();
        var visitSeriesRepository = scope.ServiceProvider.GetRequiredService<IVisitSeriesRepository>();
        
        var cutoffDate = DateTime.UtcNow.AddDays(-_autoEndDaysThreshold);
        
        // 优化查询：使用索引和分页处理
        var expiredSeries = await visitSeriesRepository
            .GetActiveSeriesOlderThanAsync(cutoffDate, pageSize: 100);
            
        if (!expiredSeries.Any())
        {
            _logger.LogDebug("未找到需要自动结束的就诊系列");
            return;
        }
        
        _logger.LogInformation("找到 {Count} 个需要自动结束的就诊系列", expiredSeries.Count);
        
        foreach (var series in expiredSeries)
        {
            try
            {
                series.Status = VisitSeriesStatus.Ended;
                series.EndDate = DateTime.UtcNow;
                series.EndReason = $"系统自动结束：超过{_autoEndDaysThreshold}天未活动";
                
                await visitSeriesRepository.UpdateAsync(series);
                
                _logger.LogInformation("自动结束就诊系列 {SeriesId}，患者ID：{PatientId}，租户ID：{TenantId}", 
                    series.Id, series.PatientId, series.TenantId);
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "自动结束就诊系列 {SeriesId} 时发生错误", series.Id);
            }
        }
    }
}

### 2.4 AI医案生成模块

#### 2.4.1 模块概述

AI医案生成模块是系统的重要特色功能，旨在基于患者的就诊系列自动生成标准化的中医医案。该模块通过分析就诊系列中的所有就诊记录，生成包含证候演变分析、治疗效果评估和诊疗建议的综合性医案报告。

#### 2.4.2 核心流程

1. **触发条件**：当医生保存就诊记录后，系统自动检测是否满足AI医案生成条件
2. **数据收集**：收集患者基本信息和当前就诊系列的所有就诊记录
3. **AI分析**：调用AI服务分析证候演变、治疗效果和诊疗建议
4. **医案生成**：生成结构化的AI医案内容
5. **医生审核**：医生审核、修改和确认AI医案
6. **版本管理**：记录医案的修改历史和版本信息

#### 2.4.3 类设计

```csharp
public class AiMedicalRecordService : IAiMedicalRecordService
{
    private readonly IAiMedicalRecordRepository _recordRepository;
    private readonly IVisitSeriesRepository _visitSeriesRepository;
    private readonly IVisitRepository _visitRepository;
    private readonly IAiAnalysisService _aiAnalysisService;
    private readonly ITenantContext _tenantContext;
    private readonly ICurrentUser _currentUser;
    private readonly ILogger<AiMedicalRecordService> _logger;
    
    public async Task<AiMedicalRecordDto> GenerateRecordAsync(GenerateRecordRequest request)
    {
        // 验证就诊系列存在且属于当前租户
        var visitSeries = await _visitSeriesRepository.GetByIdAsync(request.VisitSeriesId);
        if (visitSeries == null || visitSeries.TenantId != _tenantContext.TenantId)
            throw new BusinessException("就诊系列不存在");
            
        // 检查是否已存在AI医案
        var existingRecord = await _recordRepository.GetByVisitSeriesIdAsync(request.VisitSeriesId);
        if (existingRecord != null && existingRecord.Status != AiMedicalRecordStatus.Rejected)
            throw new BusinessException("该就诊系列已存在AI医案");
            
        // 收集就诊数据
        var visits = await _visitRepository.GetBySeriesIdAsync(request.VisitSeriesId);
        if (!visits.Any())
            throw new BusinessException("就诊系列中没有就诊记录");
            
        // 创建AI医案记录
        var record = new AiMedicalRecord
        {
            VisitSeriesId = request.VisitSeriesId,
            Status = AiMedicalRecordStatus.Generating,
            GeneratedByDoctorId = _currentUser.UserId,
            GeneratedAt = DateTime.UtcNow
        };
        
        await _recordRepository.AddAsync(record);
        
        // 异步生成AI医案内容
        _ = Task.Run(async () => await GenerateAiContentAsync(record.Id, visits));
        
        return MapToDto(record);
    }
    
    private async Task GenerateAiContentAsync(int recordId, List<Visit> visits)
    {
        try
        {
            // 构建AI分析请求
            var analysisRequest = new MedicalRecordAnalysisRequest
            {
                PatientInfo = BuildPatientInfo(visits.First()),
                VisitRecords = visits.Select(BuildVisitInfo).ToList()
            };
            
            // 调用AI服务生成医案
            var aiResult = await _aiAnalysisService.GenerateMedicalRecordAsync(analysisRequest);
            
            // 更新记录状态和内容
            var record = await _recordRepository.GetByIdAsync(recordId);
            record.Content = aiResult.Content;
            record.SyndromeEvolution = aiResult.SyndromeEvolution;
            record.TreatmentEffectiveness = aiResult.TreatmentEffectiveness;
            record.DiagnosticSuggestions = aiResult.DiagnosticSuggestions;
            record.Status = AiMedicalRecordStatus.PendingReview;
            
            await _recordRepository.UpdateAsync(record);
            
            _logger.LogInformation("AI医案生成完成，记录ID：{RecordId}", recordId);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "AI医案生成失败，记录ID：{RecordId}", recordId);
            
            // 更新记录状态为生成失败
            var record = await _recordRepository.GetByIdAsync(recordId);
            record.Status = AiMedicalRecordStatus.GenerationFailed;
            record.ErrorMessage = ex.Message;
            await _recordRepository.UpdateAsync(record);
        }
    }
    
    public async Task<bool> ConfirmRecordAsync(int recordId, ConfirmRecordRequest request)
    {
        var record = await _recordRepository.GetByIdAsync(recordId);
        if (record == null)
            throw new BusinessException("AI医案不存在");
            
        // 验证权限：只有生成医生或同租户医生可以确认
        var visitSeries = await _visitSeriesRepository.GetByIdAsync(record.VisitSeriesId);
        if (visitSeries.TenantId != _tenantContext.TenantId)
            throw new BusinessException("无权限操作此医案");
            
        if (record.Status != AiMedicalRecordStatus.PendingReview)
            throw new BusinessException("医案状态不允许确认操作");
            
        // 应用医生的修改
        if (!string.IsNullOrEmpty(request.ModifiedContent))
        {
            // 保存原始内容到历史记录
            await SaveModificationHistoryAsync(recordId, record.Content, request.ModifiedContent, request.ModificationReason);
            record.Content = request.ModifiedContent;
        }
        
        record.Status = AiMedicalRecordStatus.Confirmed;
        record.ConfirmedAt = DateTime.UtcNow;
        record.ConfirmedByDoctorId = _currentUser.UserId;
        record.Modifications = request.ModificationReason;
        
        await _recordRepository.UpdateAsync(record);
        return true;
    }
    
    public async Task<bool> RegenerateRecordAsync(int recordId, RegenerateRecordRequest request)
    {
        var record = await _recordRepository.GetByIdAsync(recordId);
        if (record == null)
            throw new BusinessException("AI医案不存在");
            
        // 验证权限
        var visitSeries = await _visitSeriesRepository.GetByIdAsync(record.VisitSeriesId);
        if (visitSeries.TenantId != _tenantContext.TenantId)
            throw new BusinessException("无权限操作此医案");
            
        // 创建新版本记录
        var newRecord = new AiMedicalRecord
        {
            VisitSeriesId = record.VisitSeriesId,
            Status = AiMedicalRecordStatus.Generating,
            GeneratedByDoctorId = _currentUser.UserId,
            GeneratedAt = DateTime.UtcNow,
            PreviousVersionId = recordId,
            RegenerationReason = request.Reason
        };
        
        await _recordRepository.AddAsync(newRecord);
        
        // 标记旧记录为已替换
        record.Status = AiMedicalRecordStatus.Replaced;
        record.ReplacedByRecordId = newRecord.Id;
        await _recordRepository.UpdateAsync(record);
        
        // 异步生成新的AI医案内容
        var visits = await _visitRepository.GetBySeriesIdAsync(record.VisitSeriesId);
        _ = Task.Run(async () => await GenerateAiContentAsync(newRecord.Id, visits));
        
        return true;
    }
}

public enum AiMedicalRecordStatus
{
    Generating = 0,           // 生成中
    PendingReview = 1,        // 待审核
    Confirmed = 2,            // 已确认
    Rejected = 3,             // 已拒绝
    GenerationFailed = 4,     // 生成失败
    Replaced = 5              // 已被新版本替换
}

    public DateTime? ConfirmedAt { get; set; }
    public int? ConfirmedByDoctorId { get; set; }
    public string ConfirmedByDoctorName { get; set; }

// /Doctor/Pages/Visits/Create.cshtml.cs
public class CreateModel : PageModel
{
    private readonly IVisitSeriesService _visitSeriesService;
    private readonly IPatientService _patientService;
    
    [BindProperty]
    public CreateVisitRequest VisitRequest { get; set; }
    
    [BindProperty]
    public bool ForceCreateNewSeries { get; set; }
    
    public VisitSeriesDto ExistingActiveSeries { get; set; }
    public bool HasActiveSeries { get; set; }
    
    public async Task<IActionResult> OnGetAsync(int patientId, int? seriesId = null)
    {
        // 验证患者存在且属于当前租户
        var patient = await _patientService.GetPatientByIdAsync(patientId);
        if (patient == null)
            return NotFound();
            
        VisitRequest = new CreateVisitRequest { PatientId = patientId };
        
        if (seriesId.HasValue)
        {
            // 继续现有就诊系列
            VisitRequest.SeriesId = seriesId.Value;
        }
        else
        {
            // 检查是否存在活跃的就诊系列
            ExistingActiveSeries = await _visitSeriesService.GetActiveSeriesByPatientAsync(patientId);
            HasActiveSeries = ExistingActiveSeries != null;
        }
        
        return Page();
    }
    
    public async Task<IActionResult> OnPostAsync()
    {
        if (!ModelState.IsValid)
            return Page();
            
        try
        {
            if (VisitRequest.SeriesId == null && ForceCreateNewSeries)
            {
                // 强制创建新系列，后端会自动结束旧系列
                VisitRequest.CreateNewSeries = true;
            }
            
            var result = await _visitService.CreateVisitAsync(VisitRequest);
            return RedirectToPage("./Details", new { id = result.Id });
        }
        catch (BusinessException ex)
        {
            ModelState.AddModelError("", ex.Message);
            return Page();
        }
    }
}
