**所有模块必须100%使用此统一状态枚举，禁止自定义状态值**。

```csharp
// 文件路径：src/Domain/Enums/TreatmentStatus.cs
/// <summary>
/// 治疗方案全生命周期状态枚举（系统级强制标准）
/// 覆盖计划1-7所有模块的状态流转需求
/// </summary>
public enum TreatmentStatus
{
    /// <summary>
    /// AI已生成治疗方案，等待医生编辑（计划2完成状态）
    /// 前置条件：计划2 AI生成完成
    /// 后置流转：计划3医生编辑开始
    /// </summary>
    AIGenerated = 1,
    
    /// <summary>
    /// 医生正在编辑治疗方案（计划3执行状态）
    /// 前置条件：计划2状态为AIGenerated
    /// 后置流转：计划4配伍禁忌检查
    /// </summary>
    Editing = 2,
    
    /// <summary>
    /// 正在进行配伍禁忌检查（计划4执行状态）
    /// 前置条件：计划3编辑完成
    /// 后置流转：检查通过→Versioning，失败→CheckFailed
    /// </summary>
    Checking = 3,
    
    /// <summary>
    /// 配伍禁忌检查失败，需返回编辑（计划4失败状态）
    /// 前置条件：计划4检查发现风险
    /// 后置流转：返回计划3重新编辑
    /// </summary>
    CheckFailed = 4,
    
    /// <summary>
    /// 正在进行版本管理（计划5执行状态）
    /// 前置条件：计划4检查通过
    /// 后置流转：版本生成完成→Finalized
    /// </summary>
    Versioning = 5,
    
    /// <summary>
    /// 治疗方案已最终确认，可打印导出（计划5完成状态）
    /// 前置条件：计划5版本管理完成
    /// 后置流转：计划7打印导出
    /// </summary>
    Finalized = 6,
    
    /// <summary>
    /// 治疗方案已归档（历史版本保留）
    /// 前置条件：新版本生成后，旧版本自动归档
    /// 后置流转：仅用于历史查询，不可再编辑
    /// </summary>
    Archived = 7
}

/// <summary>
/// 审核状态枚举（用于四级验证机制）
/// </summary>

    Pending = 1,    // 待审核
    Approved = 2,   // 审核通过
    Rejected = 3,   // 审核拒绝
```

#### 2.0.2 版本号统一规范（强制标准）
**所有版本字段必须使用三段式格式：V{主版本}.{次版本}.{修订号}**

```csharp
// 版本号生成规则（与计划5完全统一）
public class VersionNumberGenerator
{
    public static string GenerateInitialVersion() => "V1.0.0";
    public static string GenerateNextVersion(string currentVersion, VersionChangeType changeType)
    {
        var parts = currentVersion.TrimStart('V').Split('.');
        var major = int.Parse(parts[0]);
        var minor = int.Parse(parts[1]);
        var patch = int.Parse(parts[2]);
        
        return changeType switch
        {
            VersionChangeType.Major => $"V{major + 1}.0.0",
            VersionChangeType.Minor => $"V{major}.{minor + 1}.0",
            VersionChangeType.Patch => $"V{major}.{minor}.{patch + 1}",
            _ => currentVersion
        };
    }
}

public enum VersionChangeType
{
    Major, // 重大架构变更
    Minor, // 内容调整
    Patch  // 细节优化
}
```

```sql
CREATE TABLE DoctorVerificationRecords (

    EntityType VARCHAR(100) NOT NULL,
    EntityId BIGINT NOT NULL,
    DoctorId BIGINT NOT NULL,
    DoctorName VARCHAR(100) NOT NULL,
    RiskLevel ENUM('Low', 'Medium', 'High', 'Critical') NOT NULL,
```
